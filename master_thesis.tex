\begin{algorithm}[p]
\caption{Trace Selection Algorithm for calculating a template in profiling phase}
\label{alg:TraceSelection}
\begin{algorithmic}[1]
\REQUIRE power consumption traces $t_{v,1},t_{v,2},...,t_{v,n}\in \mathbb{R}^{p\times 1}$, rate $\alpha$
\ENSURE $t_{v,1^{*}},t_{v,2^{*}},...,t_{v,r^{*}}$
\STATE If $\alpha=0$, return $t_{v,1},t_{v,2},...,t_{v,n}$
\STATE For each $t_{v,i}$, calculate template $T_{v,i}$ based on $\{t_{v,1},t_{v,2},...,t_{v,n}\}\setminus \{t_{v,i}\}$
\STATE Calculate ordered pair set $\{(t_{v,i},p_{v,i}=P(t_{v,i}|T_{v,i}))|i=1,2,...,n\}$
\STATE Sort $\{(t_{v,i},p_{v,i})\}$ based on second coordinate in decreasing order and get $\{(t_{v,i^{*}},p_{v,i^{*}})\}$
\STATE Select $t_{v,1^{*}},t_{v,2^{*}},...,t_{v,r^{*}}$ where $r^{*}=n\times \frac {100-\alpha} {100}$
\end{algorithmic}
\end{algorithm}
